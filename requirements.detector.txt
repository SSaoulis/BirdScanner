# Pin numpy/onnxruntime to the exact versions proven on the Pi host. numpy
# 1.24.2 is what the apt-built picamera2 / simplejpeg / opencv stack on
# Raspberry Pi OS Bookworm is compiled against; pinning it here stops pip from
# pulling numpy 2.x into /usr/local, which would shadow the apt numpy and crash
# the camera stack with "numpy.dtype size changed". onnxruntime 1.23.2 is the
# version known to run against numpy 1.24.2 on this Pi.
numpy==1.24.2
# NB: OpenCV is intentionally NOT pip-installed here. It comes from apt
# (python3-opencv) in Dockerfile.detector so its C-ABI matches the apt-built
# picamera2/simplejpeg stack. A pip opencv-python-headless would drag in a
# conflicting numpy and shadow the apt copy, crashing the camera stack with
# "numpy.dtype size changed". (It also has no ARM wheel for a bare 4.7.0 pin.)
onnxruntime==1.23.2
# Pinned exactly (not `>=`) so pip does not backtrack through sqlmodel/pydantic
# releases on the Pi's slow link.
sqlmodel==0.0.38
